js code

import math

def circlePoints(xc, yc, R, N):
  M = round(360 / N);

  xx = []
  yy = []
  for i in range(N):
    q = i * (math.pi / 180) * M
    x = xc + R * math.cos(q)
    y = yc + R * math.sin(q)
    xx.append(x)
    yy.append(y)    
  
  return [xx, yy]

N1 = 20
[x1, y1] = circlePoints(3, 3, 2, N1)
for i in range(N1):
  print(
    f'{x1[i]:.3f}', 
    f'{y1[i]:.3f}',
    sep=','
  )

print()
