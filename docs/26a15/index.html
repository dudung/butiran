:root {
  color-scheme: light dark;

  /* Light theme */
  --bg: #ffffff;
  --fg: #222222;
  --border: #222222;
  --muted-border: #dddddd;
  --link: #0000ee;
  --box-bg: #ffffff;
  --accent: #4f81bd;
  --hover-bg: #f0f0f0;

  /* Scrollbar (Firefox) */
  scrollbar-color: var(--accent) var(--hover-bg);

  /* Form / JS-created elements */
  --control-bg: var(--box-bg);
  --control-fg: var(--fg);
  --control-border: var(--border);
}

/* Dark theme overrides */
@media (prefers-color-scheme: dark) {
  :root {
    --bg: #0d1117;
    --fg: #c9d1d9;
    --border: #30363d;
    --muted-border: #30363d;
    --link: #58a6ff;
    --box-bg: #161b22;
    --accent: #58a6ff;
    --hover-bg: #21262d;

    /* Scrollbar (Firefox) */
    scrollbar-color: var(--accent) var(--hover-bg);
    
    /* Form / JS-created elements */
    --control-bg: var(--box-bg);
    --control-fg: var(--fg);
    --control-border: var(--border);
  }

  .logo-container svg {
    filter: brightness(0.8) saturate(0.85);
  }  
}

/* JS elements */
canvas,
textarea,
input,
select {
  background-color: var(--control-bg);
  color: var(--control-fg);
  border: 1px solid var(--control-border);
}
