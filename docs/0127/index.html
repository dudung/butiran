import numpy as np
from numpy import linalg as la

C = np.array(
    [
      [2.682, 2.074, 0.858],
      [2.074, 1.668, 0.856],
      [0.858, 0.856, 0.852],
      ]
  )

print("C =",)
print(C)
print()

print("CE = EL")
L, E = la.eig(C)

print("E =")
print(E.round(3))
print()

print("L =")
print(np.diag(L).round(3))
print()

print("C @ E =")
print((C @ E).round(3))
print()

print("E @ L =")
print((E @ np.diag(L)).round(3))
print()
