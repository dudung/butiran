function button_click() {
  var x = document.getElementById("x");
  var op = document.getElementById("op"); 
  var y = document.getElementById("y");
  var z = document.getElementById("z");

  var expr = x.value + op.value + y.value;
  z.value = eval(expr);
}
</script>
