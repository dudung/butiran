import random as rnd

steps = 100
Nx = [0] * (steps + 1)
x = []
for i in range(-steps//2, steps//2 + 1):
  x.append(i)

repetition = 10000
for r in range(repetition):
  y = 0
  
  for s in range(steps):
    dy = rnd.randint(0, 1) * 2 - 1
    y += dy

  i = x.index(y)
  Nx[i] += 1

print(x)
print(Nx)
