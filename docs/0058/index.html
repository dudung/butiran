import random as rnd

xmin = 3
xmax = 7
Nx = xmax - xmin + 1

for i in range(10):
  x = [0] * Nx

  Ni = 200
  for i in range(Ni):
    xi = rnd.randint(xmin, xmax)
    x[xi - xmin] += 1
  
  print(x)
