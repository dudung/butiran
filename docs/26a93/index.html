JS codes can be written directly in a Hugo post or in external JS files.

loader

{{< script/loader >}}
library-1.js
library-2.js
library-3.js
{{< /script/loader >}}

runner

{{< script/runner id="container" >}}
// define some functions
function createElement() {
  let el = document.createElement("element");
}

// get container
let cnt = document.getElementById("container");
cnt.style.marginTop = "1em";

// create elements
let el = createElement();

// attach elements to container
cnt.append(el);
{{< /script/runner >}}
