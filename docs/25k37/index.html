Create new Hugo post, add it to Git, and open it with CotEditor.

#!/bin/bash

if [ -z "$1" ]; then
  echo "Usage: $0 <path/to/file/post-title.md>"
  exit 1
fi

POST_PATH="$1"

# Create new post using Hugo
hugo new "$POST_PATH"

# Add the post to Git
git add "$POST_PATH"

# Open the post in CotEditor
cot "$POST_PATH"
